18 /*
19  * SYNOPSIS
20  * #include <sys_lib.h>
21  *
22  * int readline( FILE *pfin, char *pbuff, int size );
23  * *pfin - pointer to file you're reading from.
24  * *pbuff - character array into which character are being read.
25  * size - The size of the buffer.
26  * (counted from 1, ie. 1 .. size).
27  *
28  * DESCRIPTION
29  *
30  * This function attempts to read in a block of data of length "size"
31  * form the file pointed to by "pfile". The read operation is done
32  * from the current position of the file pointer (in the file). If
33  * there are fewer characters than "size" left on the line (a line is
34  * a string of characters terminated by a "/n" or an EOF character)
35  * then only those characters will be read in.
36  *
37  * The integer value returned by the function will be equal to the
38  * number of characters read in unless readline is called when the
39  * file pointer is pointing directly at "EOF" then the function
40  * will return a -1 (the system constant "ERROR" ).
41  *
42  * The string of characters returned in pbuff will always be
43  * terminated by a NULL character, "\O". The first NULL character
44  * in pbuff indicates the end of the string that was read in.
45  * Be forewarned there may be more than one NULL character in pbuff,
46  * but it is only the first one that is significant.
47  * Any characters returned in the buffer after the first NULL should
48  * be ignored.
49  *
50  * If no characters are read in then the first character in pbuff
51  * will be the NULL character and the function will return a value
52  * of -1 (the system constant, "ERROR" ).
53  *
54  * RETURN
55  * ERROR size <= 0
56  * ERROR EOF
57  * #char String length.
58  *
59  * BUGS
60  * If the value of size is selected such that it is less than the
61  * number of characters on the line pointed to by pfile then the next
62  * time readline is called it will continue by grabbing characters
63  * from the same line.
64  * So it is possible that many "readline"s could be called on the same
65  * line before all the characters on the line are exhausted.
66  */
67 extern int readline(FILE *pfile, char *pbuff, int size);
